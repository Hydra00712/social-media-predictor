def calculate_engagement(likes, comments, shares, views):
    if views == 0:
        return 0.0
    total_interactions = likes + comments + shares
    engagement_rate = total_interactions / views
    return min(max(engagement_rate, 0), 1)  # Clamp to [0, 1]

def calculate_toxicity(profanity, hate_speech, harassment, spam, offensive):
    """
    Calculate toxicity score from multiple indicators
    All inputs should be between 0.0 and 1.0
    """
    weights = {
        'profanity': 0.2,
        'hate_speech': 0.3,
        'harassment': 0.25,
        'spam': 0.15,
        'offensive': 0.1
    }
    
    toxicity = (
        profanity * weights['profanity'] +
        hate_speech * weights['hate_speech'] +
        harassment * weights['harassment'] +
        spam * weights['spam'] +
        offensive * weights['offensive']
    )
    
    # Clamp to [0, 1] range
    return min(max(toxicity, 0), 1)

# Result: 0.055 or 5.5%

def calculate_sentiment(positive_count, negative_count, neutral_count, emoji_score):
    """
    Calculate sentiment score using word-based analysis
    Returns: float between -1.0 and +1.0
    """
    total = positive_count + negative_count + neutral_count + 1
    
    # Raw sentiment calculation
    raw_sentiment = ((positive_count - negative_count) / total) * 2 - 1
    
    # Adjust with emoji sentiment (weighted at 0.3)
    adjusted = raw_sentiment + (emoji_score * 0.3)
    
    # Clamp to [-1, +1] range
    return min(max(adjusted, -1), 1)

def calculate_buzz_change(current_buzz, previous_buzz, sentiment_score):
    """
    Calculate buzz change rate with optional sentiment adjustment
    """
    if previous_buzz == 0:
        return 0.0
    
    # Calculate basic buzz change percentage
    buzz_change = ((current_buzz - previous_buzz) / previous_buzz) * 100
    
    # Apply sentiment multiplier
    if sentiment_score > 0.5:
        multiplier = 1.2  # Boost positive sentiment
    elif sentiment_score < -0.5:
        multiplier = 0.8  # Dampen negative sentiment
    else:
        multiplier = 1.0  # Neutral
    
    # Calculate adjusted buzz change
    adjusted_buzz = buzz_change * multiplier
    
    # Clamp to reasonable range [-100, 100]
    return min(max(adjusted_buzz, -100), 100)

import numpy as np
from typing import Tuple

class MetricCalculator:
    """
    Complete implementation of all metric formulas
    """
    
    @staticmethod
    def calculate_engagement(likes: int, comments: int, shares: int, views: int) -> float:
        """
        Calculate engagement rate
        
        Args:
            likes: Number of likes
            comments: Number of comments
            shares: Number of shares/retweets
            views: Total number of views
            
        Returns:
            Engagement rate (0.0 to 1.0)
        """
        if views == 0:
            return 0.0
        total_interactions = likes + comments + shares
        engagement_rate = total_interactions / views
        return np.clip(engagement_rate, 0, 1)
    
    @staticmethod
    def calculate_toxicity(profanity: float, hate_speech: float, 
                          harassment: float, spam: float, offensive: float) -> float:
        """
        Calculate toxicity score from multiple indicators
        
        Args:
            profanity: Profanity score (0.0 to 1.0)
            hate_speech: Hate speech score (0.0 to 1.0)
            harassment: Harassment score (0.0 to 1.0)
            spam: Spam probability (0.0 to 1.0)
            offensive: Offensive language score (0.0 to 1.0)
            
        Returns:
            Toxicity score (0.0 to 1.0)
        """
        toxicity = (
            profanity * 0.2 +
            hate_speech * 0.3 +
            harassment * 0.25 +
            spam * 0.15 +
            offensive * 0.1
        )
        return np.clip(toxicity, 0, 1)
    
    @staticmethod
    def calculate_sentiment(positive_count: int, negative_count: int, 
                           neutral_count: int, emoji_score: float) -> float:
        """
        Calculate sentiment score using word-based analysis
        
        Args:
            positive_count: Number of positive words
            negative_count: Number of negative words
            neutral_count: Number of neutral words
            emoji_score: Emoji sentiment (-1.0 to 1.0)
            
        Returns:
            Sentiment score (-1.0 to 1.0)
        """
        total = positive_count + negative_count + neutral_count + 1
        raw_sentiment = ((positive_count - negative_count) / total) * 2 - 1
        adjusted = raw_sentiment + (emoji_score * 0.3)
        return np.clip(adjusted, -1, 1)
    
    @staticmethod
    def calculate_buzz_change(current_buzz: float, previous_buzz: float, 
                             sentiment_score: float) -> float:
        """
        Calculate buzz change rate with sentiment adjustment
        
        Args:
            current_buzz: Current engagement/buzz metric
            previous_buzz: Previous engagement/buzz metric
            sentiment_score: Sentiment score (-1.0 to 1.0)
            
        Returns:
            Buzz change percentage (-100 to 100)
        """
        if previous_buzz == 0:
            return 0.0
        
        buzz_change = ((current_buzz - previous_buzz) / previous_buzz) * 100
        
        # Apply sentiment multiplier
        if sentiment_score > 0.5:
            multiplier = 1.2
        elif sentiment_score < -0.5:
            multiplier = 0.8
        else:
            multiplier = 1.0
        
        adjusted_buzz = buzz_change * multiplier
        return np.clip(adjusted_buzz, -100, 100)
